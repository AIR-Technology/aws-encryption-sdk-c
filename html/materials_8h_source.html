264 AWS_CRYPTOSDK_STATIC_INLINE bool aws_cryptosdk_private_refcount_down(struct aws_atomic_var *refcount) {
265  /*
266  * Memory ordering note: We must use release_acquire memory order here. Otherwise, we have the following race:
267  *
268  * Program order:
269  *
270  * Thread A:
271  * Release(obj) [if down() { free(obj) } ]
272  *
273  * Thread B:
274  * obj->foo = 1;
275  * Release(obj)
276  *
277  * Execution order:
278  *
279  * Thread B: down() -> false
280  * Thread A: down() -> true
281  * Thread A: free(obj)
282  * Thread B: obj->foo = 1
283  *
284  * To prevent this we use release_acquire order. The release forbids any memory accesses sequenced-before the
285  * atomic decrement of down() from being reordered later. This prevents thread B from reordering the obj->foo
286  * access to come after the down(). The acquire ensures that the atomic decrements of down() calls correctly
287  * synchronize-with one-another.
288  */
289  size_t old_count = aws_atomic_fetch_sub_explicit(refcount, 1, aws_memory_order_acq_rel);
290 
291  assert(old_count != 0);
292 
293  return old_count == 1;
294 }

299 AWS_CRYPTOSDK_STATIC_INLINE void aws_cryptosdk_private_refcount_up(struct aws_atomic_var *refcount) {
300  /*
301  * Memory ordering note: It is safe to use relaxed here. As an invariant, we assume that,
302  * on entry, the thread has some guarantee that the refcount will not reach zero until after
303  * refcount_up completes. As long as this guarantee is maintained until the next release barrier,
304  * there is no problem; if we down() on the current thread, we achieve this explicitly. Otherwise,
305  * if we communicate to some other thread that it is safe to release the reference, that communication
306  * needs to be in release order, as it's effectively a proxy down() call, for the same reasons outlined
307  * above in refcount_down().
308  *
309  * Since we've established that our initial reference won't be released until a release barrier
310  * occurs, we are happy to let the increment be deferred until that barrier occurs. We're also
311  * perfectly happy with a refcount increment happening logically earlier than expected, since this
312  * won't cause the object to be freed unexpectedly.
313  *
314  * We also note that acquire order is not required: We know that the object itself is already valid
315  * when we enter refcount_up - that is, refcount_up does not change the internal state of the object.
316  * Therefore, it's okay if some code accesses the logical state of the object "before" the refcount
317  * increment, so long as the object is not actually freed.
318  */
319  size_t old_count = aws_atomic_fetch_add_explicit(refcount, 1, aws_memory_order_relaxed);
320 
321  assert(old_count != 0 && old_count != SIZE_MAX);
322 
323  // Suppress unused variable warning when NDEBUG is set
324  (void)old_count;
325 }
